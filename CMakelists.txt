cmake_minimum_required(VERSION 3.10)
project(MySnake)

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# 指定 SDL2 路径 (Windows MinGW 64位)
set(SDL2_PATH "${CMAKE_SOURCE_DIR}/third_party/SDL2-2.30.0/x86_64-w64-mingw32")
set(SDL2_IMAGE_PATH "${CMAKE_SOURCE_DIR}/third_party/SDL2_image-2.8.10/x86_64-w64-mingw32")
set(SDL2_MIXER_PATH "${CMAKE_SOURCE_DIR}/third_party/SDL2_mixer-2.8.1/x86_64-w64-mingw32")
set(SDL2_TTF_PATH "${CMAKE_SOURCE_DIR}/third_party/SDL2_ttf-2.24.0/x86_64-w64-mingw32")

# 添加头文件搜索路径
include_directories("${SDL2_PATH}/include" "${SDL2_PATH}/include/SDL2")
include_directories("${SDL2_IMAGE_PATH}/include" "${SDL2_IMAGE_PATH}/include/SDL2")
include_directories("${SDL2_MIXER_PATH}/include" "${SDL2_MIXER_PATH}/include/SDL2")
include_directories("${SDL2_TTF_PATH}/include" "${SDL2_TTF_PATH}/include/SDL2")
include_directories("${CMAKE_SOURCE_DIR}/srcs")

# 添加库文件搜索路径
link_directories("${SDL2_PATH}/lib")
link_directories("${SDL2_IMAGE_PATH}/lib")
link_directories("${SDL2_MIXER_PATH}/lib")
link_directories("${SDL2_TTF_PATH}/lib")

# 收集 srcs 目录下所有的源文件
file(GLOB SOURCES "srcs/*.cpp" "srcs/*.c")

# 如果没有源文件（尚未编写），CMake 会报错。我们确保至少有一个 main 文件被构建。
add_executable(${PROJECT_NAME} ${SOURCES})

# 链接相关的 SDL2 库
# MinGW 环境下通常先链接基础 SDL2，再链接扩展库
target_link_libraries(${PROJECT_NAME} mingw32 SDL2main SDL2 SDL2_image SDL2_mixer SDL2_ttf)

# 构建完成后将 SDL2.dll 复制到可执行文件目录，避免运行时缺少 DLL
set(SDL2_DLL_PATH "${SDL2_PATH}/bin/SDL2.dll")
if(EXISTS "${SDL2_DLL_PATH}")
	add_custom_command(TARGET ${PROJECT_NAME} POST_BUILD
		COMMAND ${CMAKE_COMMAND} -E copy_if_different
			"${SDL2_DLL_PATH}"
			"$<TARGET_FILE_DIR:${PROJECT_NAME}>"
	)
else()
	message(WARNING "SDL2.dll not found at: ${SDL2_DLL_PATH}")
endif()

set(SDL2_IMAGE_DLL_PATH "${SDL2_IMAGE_PATH}/bin/SDL2_image.dll")
if(EXISTS "${SDL2_IMAGE_DLL_PATH}")
	add_custom_command(TARGET ${PROJECT_NAME} POST_BUILD
		COMMAND ${CMAKE_COMMAND} -E copy_if_different
			"${SDL2_IMAGE_DLL_PATH}"
			"$<TARGET_FILE_DIR:${PROJECT_NAME}>"
	)
else()
	message(WARNING "SDL2_image.dll not found at: ${SDL2_IMAGE_DLL_PATH}")
endif()

set(SDL2_MIXER_DLL_PATH "${SDL2_MIXER_PATH}/bin/SDL2_mixer.dll")
if(EXISTS "${SDL2_MIXER_DLL_PATH}")
	add_custom_command(TARGET ${PROJECT_NAME} POST_BUILD
		COMMAND ${CMAKE_COMMAND} -E copy_if_different
			"${SDL2_MIXER_DLL_PATH}"
			"$<TARGET_FILE_DIR:${PROJECT_NAME}>"
	)
else()
	message(WARNING "SDL2_mixer.dll not found at: ${SDL2_MIXER_DLL_PATH}")
endif()

set(SDL2_TTF_DLL_PATH "${SDL2_TTF_PATH}/bin/SDL2_ttf.dll")
if(EXISTS "${SDL2_TTF_DLL_PATH}")
	add_custom_command(TARGET ${PROJECT_NAME} POST_BUILD
		COMMAND ${CMAKE_COMMAND} -E copy_if_different
			"${SDL2_TTF_DLL_PATH}"
			"$<TARGET_FILE_DIR:${PROJECT_NAME}>"
	)
else()
	message(WARNING "SDL2_ttf.dll not found at: ${SDL2_TTF_DLL_PATH}")
endif()
